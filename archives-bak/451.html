from datetime import datetime
from elasticsearch import Elasticsearch

连接Elastic
es = Elasticsearch()

定义索引
index_setting = {
"setting":{"number_of_shards":5,"number_of_replicas":1},
"mappings":{"doc_name_1":{"properties":
{"id":{"type":"integer"},
"name":{"type":"string"},
"birthday":{"type":"date"}}
}}
}

提交创建请求
es.indices.create(index="my-index",body=index_setting,ignore=400)
{u’acknowledged’: True}

删除索引
es.indices.delete(index=”my-index”)

添加mapping
doc_infos = {
"doc_name_2":{"properties":
{"id":{"type":"integer"},
"name":{"type":"string"},
"birthday":{"type":"date"}}
}
}
es.indices.put_mapping(index="my-index",doc_type="doc_name_2",body=doc_infos)

mapping中的fields一旦创建，不能单独删除和修改！

添加文档
es.index(index="my-index", doc_type="doc_name_2", id=42, body={"id": 1, "birthday": datetime.now(),"name":"hi_kitty"})

数据获取
es.get(index=”my-index”, doc_type=”test-type”, id=42)[‘_source’]
{u’any’: u’data’, u’timestamp’: u’2013-05-12T19:45:31.804229′}

数据查询
result = es.search(index="index_name","doc_type"="doc_name",body={"query":{"match_all":{}}},size=10000)
infos = result['hits']['hits']
for m in range(len(infos)):
print infos[m]['_id']

删除数据
es 根据id删除数据；所以在删除前，要获取到数据的_id(如上面的search)
es.delete(index="index_name","doc_type"="doc_name",id=infos[m]['_id'])

文档创建后，很难删除；一般修改文档内容通过version来实现。
