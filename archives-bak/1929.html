Dovecot
admin 2017/10/06 2017-10-06

安装配置
安装
yum -y install dovecot
配置
vim /etc/dovecot/dovecot.conf
#这里设定的服务启动后会Listen port，范例imap(143)、POP3(110)
protocols = imap pop3 lmtp

log_path = /var/log/dovecot-deliver.log
info_log_path = /var/log/dovecot-deliver.log
vim /etc/dovecot/conf.d/10-auth.conf
#取消明码传输, 若无取消在MUA端预设禁止明码会出现error。
disable_plaintext_auth = no
#用户格式
auth_username_format = %Ln
vim /etc/dovecot/conf.d/10-mail.conf
#用maildir储存格式并放到使用者的家目录~
mail_location = mbox:~/mail:INBOX=/var/mail/%u

#验证密码方式
passdb {
driver = pam
}
#验证用户方式
userdb {
driver = passwd
}

完整配置文件
protocols = imap imaps pop3 pop3s
listen = *

disable_plaintext_auth = no

log_path = /var/log/dovecot-deliver.log
info_log_path = /var/log/dovecot-deliver.log

log_timestamp = “%Y-%m-%d %H:%M:%S ”

login_user = postfix

login_greeting = Dovecot IMAP server!

mail_location = maildir:~/Maildir

mail_privileged_group = mail

mail_debug = yes
verbose_proctitle = yes

protocol imap {
login_executable = /usr/lib/dovecot/imap-login
mail_executable = /usr/lib/dovecot/imap
imap_client_workarounds = outlook-idle delay-newmail netscape-eoh tb-extra-mailbox-sep
}

protocol pop3 {
pop3_uidl_format = %08Xu%08Xv
}

protocol managesieve {
}

protocol lda {
postmaster_address =
hostname = ubuntuserver
sendmail_path = /usr/lib/sendmail

rejection_subject = Rejected: %s
rejection_reason = Your message to <%t> was automatically rejected:%n%r

auth_socket_path = /var/run/dovecot/auth-master
}

auth_username_format = %Lu
auth_verbose = yes
auth_debug = yes
auth_debug_passwords = yes

auth default {
mechanisms = plain

passdb pam {
}

userdb passwd {
}

user = root

socket listen {
master {
path = /var/run/dovecot/auth-master
mode = 0660
user = mail
group = mail
}
client {
path = /var/spool/postfix/private/auth-client
mode = 0660
user = postfix
group = postfix
}
}
!include_try /etc/dovecot/auth.d/*.auth
}
