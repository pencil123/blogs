MoinMoin wiki
admin 2015/10/28 2018-02-14

团队里一直没有共享知识的平台，所以就想建一个。在以前的公司里有用维基百科的经验，所以也想选一个；考虑到共享的内容涉及到安全和权限管理问题。就选用了MoinMoin。
MoinMoin是python编写的，页面和账号内容都存储在data目录以文件形式存储。
现在就开始安装了：
整个过程中需要安装的服务有nginx+uwsgi+Moin
首先NGINX安装，这很容易了。
#安装nginx 可以按照自己的想法安装；
yum install python-devel python-setuptools pcre pcre-devel
wget http://${nginxurlsorce}/nginx-1.2.4.tar
tar zxvf nginx-1.2.4.tar
cd nginx-1.2.4
./configure –user=nobody –group=nobody –prefix=/usr/local/nginx –with-http_stub_status_module –with-cc-opt=’-O3′ –with-cpu-opt=opteron
make && make install
#上面就完成了nginx的安装了，下面配置nginx 虚拟主机文件 uwsgi.conf
server {
listen 80;
server_name www.wiki.com;

location / {
include uwsgi_params;
uwsgi_pass 127.0.0.1:8000;
}

#接下来安装uwsgi
yum -y install libxml*
cd /opt/
wget https://pypi.python.org/packages/source/u/uWSGI/uwsgi-2.0.10.tar.gz#md5=7a9be0db5f6a8d4150dc5e9e0517ce80
tar zxvf uwsgi-2.0.10.tar.gz
cd uwsgi-2.0.10
make
cp uwsgi /usr/bin/

#最后Moin的安装
cd /opt/
yum install vim wget gcc gcc+c++ patch -y
wget http://static.moinmo.in/files/moin-1.9.8.tar.gz
tar zxvf moin-1.9.8.tar.gz
cd moin-1.9.8
#打补丁 可以省略
wget http://garfileo.is-programmer.com/user_files/garfileo/File/templates.py.patch
patch -p1 < templates.py.patch
cd /opt/moin-1.9.8
#安装Moin
python setup.py install –force –prefix /opt/wiki –record=install.log
#配置Moin
cd /opt/wiki/share/moin/
cp config/wikiconfig.py .
cp server/moin.wsgi .
cp moin.wsgi moin_wsgi.py
#配置 uwsgi.xml
cat /opt/wiki/uwsgi.xml
<uwsgi>
<socket>127.0.0.1:8000</socket>
<pythonpath>/opt/wiki/lib/python2.6/site-packages</pythonpath>
<pythonpath>/opt/wiki/share/moin</pythonpath>
<processes>9</processes>
<module>moin_wsgi</module>
<daemonize>/var/log/wiki.log</daemonize>
</uwsgi>

#修改配置文件
vim /opt/wiki/share/moin/moin.wsgi
# a1) Path of the directory where the MoinMoin code package is located.
# Needed if you installed with –prefix=PREFIX or you didn’t use setup.py.
#sys.path.insert(0, ‘PREFIX/lib/python2.3/site-packages’)
/opt/syswiki/lib/python2.6/site-packages/ # MoinMoin 安装位置根据安装目录而定义

# a2) Path of the directory where wikiconfig.py / farmconfig.py is located.
# See wiki/config/… for some sample config files.
#sys.path.insert(0, ‘/path/to/wikiconfigdir’)
#sys.path.insert(0, ‘/path/to/farmconfigdir’)
sys.path.insert(0, ‘/opt/syswiki/share/moin’) # MoinMoin 主配置文件wikiconfig.py所在目录

启动nginx
/usr/local/nginx/sbin/nginx
运行uwsgi
uwsgi -x /opt/wiki/uwsgi.xml &
