js cookie 操作
admin 2016/11/21 2017-10-25

1.清除cookie的原理：给需要删除的cookie设置expires（失效时间）,从而将删除此条数据。
代码如下：
document.cookie = name +”=data1;expires=”+(new Date()).toGMTString();
注意事项：
(1).name = data1 表示操作的是cookie中Name为data1的这条数据。
(2).关于网上有些解决方案中报错：toGMTSting() is not a function.因为toGMTString() 方法可根据格林威治时间 (GMT) 把 Date 对象转换为字符串，并返回结果。
(3).我们删除cookie的方法是用一个同样名字、过期时间为过去某个时候的Cookie覆盖之。
这时就一定要搞清楚你要删除的cookie的域和路径，Cookie域和路径要一样才能被覆盖。 否则无法删除

Cookie不仅仅有名字和值两个属性，还有域（domain），过期时间（expires），路径（path）等属性。
其中，不同的域、不同的路径下可以存在同样名字的cookie。
document.cookie = “sdmenu_” + encodeURIComponent(this.menu.id) + “=” + states.join(“”) + “; expires=” + d.toGMTString() + “; path=/”;
更新代码：
var date=new Date();
date.setTime(date.getTime()-10000);
document.cookie=name+”=v; expire=”+date.toGMTString()+”; path=/”;
除了路径（path），可能还有域等其它情况，保险的方式就是 设置Cookie时是什么样式，删除时就应该是什么样式，加上expire属性就行了

<script>
function SetCookie()//两个参数，一个是cookie的名子，一个是值
{
alert(“333”)
var name = “username”
var value = “grey”
var Days = 30; //此 cookie 将被保存 30 天
var exp = new Date(); //new Date(“December 31, 9998”);
exp.setTime(exp.getTime() + Days*24*60*60*1000);
document.cookie = name + “=”+ escape (value) + “;expires=” + exp.toGMTString();
}
function getCookie(name)//取cookies函数
{
var arr = document.cookie.match(new RegExp(“(^| )”+name+”=([^;]*)(;|$)”));
if(arr != null) return unescape(arr[2]); return null;

}
function delCookie()//删除cookie
{
var name = “username”
var exp = new Date();
exp.setTime(exp.getTime() – 1);
var cval=getCookie(name);
if(cval!=null) document.cookie= name + “=”+cval+”;expires=”+exp.toGMTString()+”;path=/”;
location.reload()
}

</script>
