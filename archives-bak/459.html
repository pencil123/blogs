数据库表的注释操作

创建表的时候写注释

create table test1
(field_name int comment '字段的注释')comment='表的注释';

修改表的注释

alter table test1 comment '修改后的表的注释';

查看表注释的方法

–在生成的SQL语句中看
show create table test1;
–在元数据的表里面看
use information_schema;
select * from TABLES where TABLE_SCHEMA='my_db' and TABLE_NAME='test1' \G

修改字段的注释

alter table test1 modify field_name int comment '修改后的字段注释';
注意：字段名和字段类型要写明

查看字段注释的方法

show full columns from test1;
–在元数据的表里面看
select * from COLUMNS where TABLE_SCHEMA='my_db' and TABLE_NAME='test1' \G

数据库表的字段列操作

查看表字段列

desc table_name；

展示表字段所有的列

show full columns from table_name;

修改表名字

alter table table_now_name rename to table_new_name

对表添加新的字段列

alter table table_name add field_name varchar(30) after column_other_name

删除字段列名

alter table table_name drop field_name;

修改字段列名

alter table table_name change column_now_name column_new_name int;

修改字段列属性

alter table table_name modify name varchar(22)
