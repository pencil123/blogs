python 使用SMTP 协议发送邮件。PS：首先需要SMTP服务器，或者第三方邮箱账号开通了smtp协议
#导入必要的类
import smtplib
import email.mime.multipart
import email.mime.text
import email.MIMEBase
import os.path
# 构造MIMEMultipart对象做为根容器
msg = email.mime.multipart.MIMEMultipart()
#对根容器属性赋值；以下信息只是用来显示，并不在发送邮件中起到具体功能性作用
msg[‘from’] = “shell@163.com”
msg[‘to’] = “123@163.com,234@163.com,345@123.com”
msg[‘subject’] = “helll world”

content = content + HtmlInfo + “</table></blockquote>”
#print(content)
# 构造MIMEText对象做为邮件显示内容并附加到根容器 第二个参数定义内容打开方式;第三个参数定义编码-防止乱码问题
txt = email.mime.text.MIMEText(content,’html’,’utf-8′)
msg.attach(txt)

# 构造MIMEBase对象做为文件附件内容并附加到根容器
contype = ‘application/octet-stream’
maintype, subtype = contype.split(‘/’, 1)
## 读入文件内容并格式化
data = open(file_name, ‘rb’)
file_msg = email.MIMEBase.MIMEBase(maintype, subtype)
file_msg.set_payload(data.read( ))
data.close( )
email.Encoders.encode_base64(file_msg)

## 设置附件头
basename = os.path.basename(file_name)
file_msg.add_header(‘Content-Disposition’,
‘attachment’, filename = basename)
msg.attach(file_msg)

mailto = [‘123@163.com’,’234@163.com’,’345@123.com’]
#多个收件人，以list格式写入
smtp=smtplib
smtp=smtplib.SMTP() #实例化对象
smtp.connect(‘mail.gogle.com’,’25’) #连接服务器
smtp.login(‘shell@mail.gogle.com’,’hillwo’) #用户登陆

smtp.sendmail(‘shell@mail.gogle.com’,mailto,str(msg))
smtp.quit()
